### 1. CONTROLLER
```php
<?php

namespace App\Http\Controllers;

use Illuminate\Http\Request;

class C_maze extends Controller
{
    public function index()
    {
        return view('labirin');
    }

    public function input_angka(Request $request)
    {
        $angka = $request->angka;
        $kosong = [];
        for ($i = 1; $i <= $angka; $i++) {
            $kosong[] = $i;
        }
        $data = [
            "angka" => $kosong,
            "max" => $angka,
            "min" => 1,
            "inputan" => $angka,
        ];
        return view('hasil_labirin', $data);
    }
}
```
